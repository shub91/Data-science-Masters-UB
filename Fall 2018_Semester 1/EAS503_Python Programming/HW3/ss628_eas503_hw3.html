def getBirthdays(pres):
    bdays = []
    N = []
    for ele in range(len(pres)):
        urlreq = requests.get(pres[ele])
        urltext = urlreq.text
        urllines = re.findall(r'<span class="bday">(.*?)</span>\)',urltext)
        urlN = re.findall(r'"wgTitle":"(.*?)"',urltext)
        bdays.append(dt.strptime(urllines[0], '%Y-%m-%d'))
        N.append(urlN[0])
    ser = pd.Series(N, index=bdays)
    return ser

import numpy as np
import time
import matplotlib.pyplot as plt
%matplotlib inline 

def QS(A):
    # Function to do quicksort
    QSR(A,0,len(A)-1)
    
def QSR(A,F,L):
    if F<L:
        split = splitf(A,F,L)
        QSR(A,F,split-1)
        QSR(A,split+1,L)
        
def splitf(A,F,L):
    P = A[F]
    LM = F+1
    RM = L
    done = False
    while not done:
        while LM <=RM and A[LM]<=P:
            LM +=1
        while A[RM] >= P and RM >=LM:
            RM -=1
        if RM < LM:
            done = True
        else:
            A[LM], A[RM] = A[RM], A[LM]
    A[F], A[RM] = A[RM], A[F]
    
    return RM

def cal_det(A):
    # LU Decomposition
    L = np.eye(len(A))
    U = A
    m = A.shape[0]
    n = A.shape[1]
    
    for i in range(m - 1):
        for j in range(i + 1, n):
            L[j][i] = U[j][i] / U[i][i]
            U[j][i] = 0
            
            for k in range(i + 1, n):
                U[j][k] -= L[j][i] * U[i][k]
    D = 1
    for i in range(m):
        D *= L[i][i] * U[i][i]
    return D

def mul_mat(X,Y):
    r = np.zeros((X.shape[0],Y.shape[1]))
    for i in range(len(X)):
        for j in range(len(Y[0])):
            for k in range(len(Y)):
                r[i][j] += X[i][k] * Y[k][j]
    return r

From the plot for Sorting, a function for Quicksort was implemented using Regular Python commands and compared to np.sort(). We observe that initially the time taken to execute both functions are close initially but the difference between them increases as the size of the array grows and the time for Regular Python commands increases for all the three plots.

This is what was expected, since the numpy functions pushes the expensive computation of for loops into lower level language i.e. C which make it more efficient. We know that the standard Python implementation is written in C and every Python object is simply a cleverly-disguised C structure which has other information as well apart from the value like reference count, type of variable, its size etc. Although, this type of implementation makes Python a very effective tool as a coding language in terms of the flexibility it offers, but this makes it slower which is apparent in the three graphs. The numpy universal functions’ implementation does all the expensive computation at a lower level thus making it more efficient.
